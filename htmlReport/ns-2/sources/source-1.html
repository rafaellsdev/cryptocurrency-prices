 fun CoroutineScope.safeLaunch(
     block: suspend CoroutineScope.() -> Unit,
     errorBlock: (error: DefaultError) -> Unit
 ) {
     launch {
         try {
             block.invoke(this)
         } catch (e: Exception) {
             errorBlock.invoke(e.toDefaultError())
         }
     }
